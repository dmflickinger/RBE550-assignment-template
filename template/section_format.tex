
% Nice section title formatting




% fancy up the section titles
\definecolor{sectioncrimson}{RGB}{171,25,45}
\newcommand{\hsp}{\hspace{8pt}}


\titleformat
{\section}
[hang]
{\normalfont\sffamily\Large\bfseries}
{}
{0pt}
{
\begin{tikzpicture}
\node[draw,fill=white,text depth=1ex,font=\Large\bfseries\sffamily] (main){\hsp#1};
\node[draw,fill=black,text depth=1ex] at (main.west){\textcolor{white}\thesection};
\end{tikzpicture}
}

% fancy up the subsection titles
\titleformat
{\subsection}
[hang]
{\normalfont\sffamily\Large\bfseries}
{}
{0pt}
{
\begin{tikzpicture}
\node[rectangle, draw, outer sep=0pt, inner sep=4pt](main){\thesubsection \hsp #1};
\end{tikzpicture}
}
